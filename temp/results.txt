SELECT 'Customers: CID --> CN' AS FD,
	CASE WHEN COUNT(*)=0 THEN 'GILDIR'
	ELSE 'gildir ekki' END AS VALIDITY
FROM (
	SELECT CID
	FROM Customers
	GROUP BY CID
	HAVING COUNT(DISTINCT CN) > 1
) X;
